# 微信分享  
```javascript
<!--------微信自定义分享-------->
//先调用微信js
<script src="//res.wx.qq.com/open/js/jweixin-1.4.0.js"></script>
//微信分享
//只能调用微信内部浏览器右上角'...' 唤起微信分享，无法做到自定义按钮调用分享接口
var urlstr = window.location.href;
var urlPath = urlstr.replace(/&/g,'%26'); 

$.ajax({
    url: k_host + "/WxGetSignature", //k_host 为本地路径，忽略就好，这个实际上就是把需要分享页面的 url 传给后台
    data:{
        url: urlPath,
    },
    async: true,
    type: 'POST',
    cache: false,
    dataType: 'json',
    success:function(data){
            var res = data.data;
            wx.config({
                debug: false, // 开启调试模式,调用的所有api的返回值会在客户端alert出来，若要查看传入的参数，可以在pc端打开，参数信息会通过log打出，仅在pc端时才会打印。
                appId: res.appid,
                timestamp: res.timestamp, // 必填，生成签名的时间戳
                nonceStr: res.noncestr, // 必填，生成签名的随机串
                signature: res.signature,// 必填，签名
                jsApiList: [ // 必填，需要使用的JS接口列表
                    'onMenuShareTimeline',
                    'onMenuShareAppMessage',
                    'onMenuShareQQ',
                    'onMenuShareWeibo',
                    'onMenuShareQZone'
                ]
            });
            
            setTimeout(()=>{
                
                wx.ready(function () {
                    wx.onMenuShareAppMessage({
                        title: '测试用 title',
                        desc: '测试用 description',
                        imgUrl: '分享图标的路径', // 分享图标
                        type: '', // 分享类型,music、video或link，不填默认为link
                        dataUrl: '', // 如果type是music或video，则要提供数据链接，默认为空
                        link: urlstr, // 分享链接，该链接域名或路径必须与当前页面对应的公众号JS安全域名一致
                        success : function () {
                            //分享成功需要执行的函数
                        },
                        cancel: function (err) {
                            // console.log(err);
                            // 用户取消分享后执行的回调函数
                        }
                    });
                    wx.onMenuShareTimeline({
                        title: '测试用 title',
                        desc: '测试用 description',
                        imgUrl: '分享图标的路径', // 分享图标
                        type: '', // 分享类型,music、video或link，不填默认为link
                        dataUrl: '', // 如果type是music或video，则要提供数据链接，默认为空
                        link: urlstr, // 分享链接，该链接域名或路径必须与当前页面对应的公众号JS安全域名一致
                        success : function () {
                            //分享成功需要执行的函数
                        },
                        cancel: function (err) {
                            console.log(err);
                            // 用户取消分享后执行的回调函数
                        }
                    });
                });
            },1000)
            wx.error(function(res){
            });
    },
    error:function(request){
        alert("连接失败，请检查网络，稍后再试");
    }
});

```
